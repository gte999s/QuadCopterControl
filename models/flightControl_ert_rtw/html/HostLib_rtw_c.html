12/* Open the library and initialize the function pointers */



13void CreateHostLibrary(const char *libName, void *hl) 



14{



15    HostLibrary *hostLib = (HostLibrary*)hl;



16    *hostLib->errorMessage = '\0';



17    memset(hostLib, 0, sizeof(HostLibrary));



18#if defined(_WIN32)



19    hostLib->library = LoadLibrary(libName);



20    if(hostLib->library == NULL)



21        sprintf(hostLib->errorMessage, "Could not open library: %s.\nTo run the generated code outside the MATLAB environment, use the packNGo function.\n", libName);



22    else {



23        hostLib->libCreate    = GetProcAddress(hostLib->library, "libCreate");



24        hostLib->libStart     = GetProcAddress(hostLib->library, "libStart"); 



25        hostLib->libReset     = GetProcAddress(hostLib->library, "libReset"); 



26        hostLib->libUpdate    = GetProcAddress(hostLib->library, "libUpdate");



27        hostLib->libOutputs   = GetProcAddress(hostLib->library, "libOutputs");



28        hostLib->libTerminate = GetProcAddress(hostLib->library, "libTerminate");



29        hostLib->libDestroy   = GetProcAddress(hostLib->library, "libDestroy");



30    }



31#elif defined(_VXWORKS_)



32    hostLib->library = dlopen(libName, RTLD_NOW);



33    if(hostLib->library == NULL) {



34        sprintf(hostLib->errorMessage, "Could not open library: %s", libName);



35    }



36    else {



37        hostLib->libCreate    = dlsym(hostLib->library, "libCreate");



38        hostLib->libStart     = dlsym(hostLib->library, "libStart"); 



39        hostLib->libReset     = dlsym(hostLib->library, "libReset"); 



40        hostLib->libUpdate    = dlsym(hostLib->library, "libUpdate");



41        hostLib->libOutputs   = dlsym(hostLib->library, "libOutputs");



42        hostLib->libTerminate = dlsym(hostLib->library, "libTerminate");



43        hostLib->libDestroy   = dlsym(hostLib->library, "libDestroy");



44    }



45#else



46    hostLib->library = dlopen(libName, RTLD_NOW | RTLD_LOCAL);



47    if(hostLib->library == NULL) {



48        sprintf(hostLib->errorMessage, "Could not open library: %s", libName);



49    }



50    else {



51        hostLib->libCreate    = dlsym(hostLib->library, "libCreate");



52        hostLib->libStart     = dlsym(hostLib->library, "libStart"); 



53        hostLib->libReset     = dlsym(hostLib->library, "libReset"); 



54        hostLib->libUpdate    = dlsym(hostLib->library, "libUpdate");



55        hostLib->libOutputs   = dlsym(hostLib->library, "libOutputs");



56        hostLib->libTerminate = dlsym(hostLib->library, "libTerminate");



57        hostLib->libDestroy   = dlsym(hostLib->library, "libDestroy");



58    }



59#endif



60    if(!*hostLib->errorMessage && 



61       (!hostLib->libCreate ||



62        !hostLib->libStart ||



63        !hostLib->libUpdate ||



64        !hostLib->libOutputs ||



65        !hostLib->libTerminate ||



66        !hostLib->libDestroy)



67        ) 



68    {



69        sprintf(hostLib->errorMessage, "Could not determine function entry points in %s", libName);



70    }



71    if(*hostLib->errorMessage) {



72        if (hostLib->library) DestroyHostLibrary(hostLib);



73        fprintf(stderr,"%s",hostLib->errorMessage);



74    }



75}



76



77/* Close the library */



78void DestroyHostLibrary(void *hl) 



79{



80    HostLibrary *hostLib = (HostLibrary*)hl;



81    if(hostLib->library) {



82#if defined(_WIN32)



83        FreeLibrary(hostLib->library);



84#else



85        dlclose(hostLib->library);



86#endif



87        /* Clear the structure (except for any possible error message). */



88		memset(hostLib, 0, sizeof(HostLibrary) - MAX_ERR_MSG_LEN);



89    }



90}



91



92void LibStart(void *hl)



93{



94    HostLibrary *hostLib = (HostLibrary*)hl;



95    if(hostLib->instance)



96        (MAKE_FCN_PTR(pFnLibStart,hostLib->libStart))(hostLib->instance, hostLib->errorMessage);



97}



98



99void LibReset(void *hl) 



100{



101    HostLibrary *hostLib = (HostLibrary*)hl;



102    if(hostLib->instance && hostLib->libReset)



103        (MAKE_FCN_PTR(pFnLibReset,hostLib->libReset))(hostLib->instance, hostLib->errorMessage);



104}



105



106void LibTerminate(void *hl) 



107{



108    HostLibrary *hostLib = (HostLibrary*)hl;



109    if(hostLib->instance)



110        (MAKE_FCN_PTR(pFnLibTerminate,hostLib->libTerminate))(hostLib->instance, hostLib->errorMessage);



111}



112



113void LibDestroy(void *hl, int type)



114{



115    HostLibrary *hostLib = (HostLibrary*)hl;



116    if(hostLib->instance)



117    {



118        (MAKE_FCN_PTR(pFnLibDestroy,hostLib->libDestroy))(hostLib->instance, hostLib->errorMessage, type);



119        hostLib->instance = NULL;



120    }



121}



122



123int LibError(void *hl) 



124{



125    HostLibrary *hostLib = (HostLibrary*)hl;



126    return hostLib->errorMessage[0] != '\0';



127}



128void PrintError(char *message)



129{



130    /* Print, and then reset, the error message */



131    printf("Error: %s\n", message);



132    message[0] = '\0';



133}



134void PrintWarning(char *message)



135{



136    /* Print, and then reset, the warning message */



137    printf("Warning: %s\n", message);



138    message[0] = '\0';



139}



140char * GetErrorBuffer(void *hl) 



141{



142    HostLibrary *hostLib = (HostLibrary*)hl;



143    return hostLib->errorMessage;



144}



145void * GetNullPointer(void) 



146{



147    return NULL;
